    find "$DIR" -maxdepth 4 -mindepth 1 -type f | while read filepath; do

* Get the parent directory for a file

  #+BEGIN_SRC bash
    filepath=/a/b/c/d/file
    parentname="$(basename "$(dirname "$filepath")")"
    echo "$parentname" # d
  #+END_SRC

  #+RESULTS:
  : d
